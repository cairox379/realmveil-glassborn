function drawHero(){ if(!heroLoaded) return; const moving = keys["a"] || keys["d"] || keys["arrowleft"] || keys["arrowright"]; /* GROUNDED WALKING BOUNCE */ const bounce = moving ? Math.abs(Math.sin(Date.now() * 0.018)) * 5 : 0; /* SLIGHT ARMOR/CLOTH SWAY */ const sway = moving ? Math.sin(Date.now() * 0.01) * 0.015 : 0; /* CAPE DRAG */ const cape = moving ? Math.sin(Date.now() * 0.012) * 4 : 0; /* SHADOW STRETCH */ const shadowStretch = moving ? 65 + Math.abs(Math.sin(Date.now() * 0.018) * 10) : 60; /* SHADOW */ ctx.save(); ctx.fillStyle = "rgba(0,0,0,0.28)"; ctx.beginPath(); ctx.ellipse( player.x + player.w * .5, player.y + player.h - 6, shadowStretch, 14, 0, 0, Math.PI * 2 ); ctx.fill(); ctx.restore(); /* CHARACTER */ ctx.save(); if(player.facing === -1){ ctx.translate( player.x + player.w, player.y + bounce ); ctx.scale(-1,1); ctx.rotate(-sway); }else{ ctx.translate( player.x, player.y + bounce ); ctx.rotate(sway); } /* CAPE DRAG OFFSET */ ctx.translate(cape * 0.08,0); /* DRAW HERO */ ctx.drawImage( hero, 0, 0, player.w, player.h ); ctx.restore(); /* SWORD TRAIL */ if(player.attack){ ctx.save(); ctx.strokeStyle = "rgba(125,247,255,.95)"; ctx.lineWidth = 7; ctx.shadowBlur = 25; ctx.shadowColor = "#7df7ff"; ctx.beginPath(); ctx.arc( player.x + 160, player.y + 160, 100, -.8, .8 ); ctx.stroke(); ctx.restore(); } /* SPELL EFFECT */ if(player.spell){ ctx.save(); const pulse = 30 + Math.sin(Date.now() * 0.01) * 6; ctx.fillStyle = "rgba(125,247,255,.55)"; ctx.shadowBlur = 45; ctx.shadowColor = "#7df7ff"; ctx.beginPath(); ctx.arc( player.x + 170, player.y + 100, pulse, 0, Math.PI * 2 ); ctx.fill(); ctx.restore(); } }
